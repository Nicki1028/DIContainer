public class ServiceProvider
{
    private readonly Dictionary<Type, object> _singletonInstances = new Dictionary<Type, object>();
    private readonly IEnumerable<ServiceDescriptor> _serviceDescriptors;

    public ServiceProvider(IEnumerable<ServiceDescriptor> serviceDescriptors)
    {
        _serviceDescriptors = serviceDescriptors;
    }

    public object GetService(Type serviceType)
    {
        // 获取相应的服务描述符
        var descriptor = _serviceDescriptors.FirstOrDefault(d => d.ServiceType == serviceType);

        if (descriptor == null)
        {
            throw new Exception($"Service of type {serviceType} is not registered.");
        }

        // 如果是 Singleton 并且实例已经存在，直接返回
        if (descriptor.Lifetime == ServiceLifetime.Singleton)
        {
            if (!_singletonInstances.TryGetValue(serviceType, out var instance))
            {
                // 如果尚未创建实例，执行 ImplementationFactory
                instance = descriptor.ImplementationFactory(this);
                _singletonInstances[serviceType] = instance; // 缓存实例
            }

            return instance;
        }

        // 如果是 Transient，每次都创建新实例
        return descriptor.ImplementationFactory(this);
    }
}
